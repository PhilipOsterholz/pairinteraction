97  // fixed)

99  Eigen::SparseMatrix<Scalar> basis_one = system_one.getBasisvectors();

103

106  .pruned(tolerance, 1) // without pruning, max_diff_hamiltonian

108  // zero

112  *std::max_element(diff.valuePtr(), diff.valuePtr() + diff.nonZeros());

116  CHECK(max_diff_hamiltonian == doctest::Approx(0.0).epsilon(tolerance));
117
