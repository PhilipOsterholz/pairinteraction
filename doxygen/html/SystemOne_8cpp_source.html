65 template <typename Scalar>
66 void SystemOne<Scalar>::setBfield(std::array<double, 3> field) {
67  this->onParameterChange();
68  bfield = field;
69 
70  // Transform the magnetic field into spherical coordinates
71  this->changeToSphericalbasis(bfield, bfield_spherical);
72 
73  diamagnetism_terms[{{0, +0}}] = bfield_spherical[+0] * bfield_spherical[+0] -
74  bfield_spherical[+1] * bfield_spherical[-1] * 2.;
75  diamagnetism_terms[{{2, +0}}] =
76  bfield_spherical[+0] * bfield_spherical[+0] + bfield_spherical[+1] * bfield_spherical[-1];
77  diamagnetism_terms[{{2, +1}}] = bfield_spherical[+0] * bfield_spherical[-1];
78  diamagnetism_terms[{{2, -1}}] = bfield_spherical[+0] * bfield_spherical[+1];
79  diamagnetism_terms[{{2, +2}}] = bfield_spherical[-1] * bfield_spherical[-1];
80  diamagnetism_terms[{{2, -2}}] = bfield_spherical[+1] * bfield_spherical[+1];
81 }

162 template <typename Scalar>
163 void SystemOne<Scalar>::initializeBasis() {
164  // If the basis is infinite, throw an error
165  if (this->range_n.empty() &&
166  (this->energy_min == std::numeric_limits<double>::lowest() ||
167  this->energy_max == std::numeric_limits<double>::max())) {
168  throw std::runtime_error(
169  "The number of basis elements is infinite. The basis has to be restricted.");
170  }
171 
172  auto &cache = *this->m_cache;
173 
174  ////////////////////////////////////////////////////////////////////
175  /// Build one atom states //////////////////////////////////////////
176  ////////////////////////////////////////////////////////////////////
177 
178  // TODO check whether symmetries are applicable
179  // TODO check whether range_j, range_m is half-integer or integer valued
180 
181  float s = 0.5;
182  if (std::isdigit(species.back()) != 0) {
183  s = ((species.back() - '0') - 1) / 2.; // TODO think of a better solution
184  }
185 
186  size_t idx = 0;
187  std::vector<Eigen::Triplet<Scalar>>
188  basisvectors_triplets; // TODO reserve states, basisvectors_triplets,
189  // hamiltonian_triplets
190 
191  std::vector<Eigen::Triplet<Scalar>> hamiltonian_triplets;
192 
193  /// Loop over specified quantum numbers ////////////////////////////
194 
195  std::set<int> range_adapted_n, range_adapted_l;
196  std::set<float> range_adapted_j, range_adapted_m;
197 
198  if (this->range_n.empty()) {
199  throw std::runtime_error(
200  "The calculation of range_n via energy restrictions is not yet implemented."); // TODO
201  }
202  range_adapted_n = this->range_n;
203 
204  for (auto n : range_adapted_n) {
205 
206  if (this->range_l.empty()) {
207  this->range(range_adapted_l, 0, n - 1);
208  } else {
209  range_adapted_l = this->range_l;
210  }
211  for (auto l : range_adapted_l) {
212  if (l > n - 1 || l < 0) {
213  continue;
214  }
215 
216  if (this->range_j.empty()) {
217  this->range(range_adapted_j, std::fabs(l - s), l + s);
218  } else {
219  range_adapted_j = this->range_j;
220  }
221  for (auto j : range_adapted_j) {
222  if (std::fabs(j - l) > s || j < 0) {
223  continue;
224  }
225 
226  double energy = StateOne(species, n, l, j, s).getEnergy(cache);
227  if (!this->checkIsEnergyValid(energy)) {
228  continue;
229  }
230 
231  if (this->range_m.empty()) {
232  this->range(range_adapted_m, -j, j);
233  } else {
234  range_adapted_m = this->range_m;
235  }
236 
237  // Consider rotation symmetry
238  std::set<float> range_allowed_m;
239  if (sym_rotation.count(static_cast<float>(ARB)) == 0) {
240  std::set_intersection(sym_rotation.begin(), sym_rotation.end(),
241  range_adapted_m.begin(), range_adapted_m.end(),
242  std::inserter(range_allowed_m, range_allowed_m.begin()));
243  } else {
244  range_allowed_m = range_adapted_m;
245  }
246 
247  for (auto m : range_allowed_m) {
248  if (std::fabs(m) > j) {
249  continue;
250  }
251 
252  // Create state
253  StateOne state(species, n, l, j, m);
254 
255  // Check whether reflection symmetry can be realized with the states available
256  if (sym_reflection != NA && state.getM() != 0 &&
257  range_allowed_m.count(-state.getM()) == 0) {
258  throw std::runtime_error("The momentum " + std::to_string(-state.getM()) +
259  " required by symmetries cannot be found.");
260  }
261 
262  // Add symmetrized basis vectors
263  this->addSymmetrizedBasisvectors(state, idx, energy, basisvectors_triplets,
264  hamiltonian_triplets, sym_reflection);
265  }
266  }
267  }
268  }
269 
270  /// Loop over user-defined states //////////////////////////////////
271 
272  // Check that the user-defined states are not already contained in the list of states
273  for (const auto &state : this->states_to_add) {
274  if (this->states.template get<1>().find(state) != this->states.template get<1>().end()) {
275  throw std::runtime_error("The state " + state.str() +
276  " is already contained in the list of states.");
277  }
278  if (!state.isArtificial() && state.getSpecies() != species) {
279  throw std::runtime_error("The state " + state.str() + " is of the wrong species.");
280  }
281  }
282 
283  // Add user-defined states
284  for (const auto &state : this->states_to_add) {
285  // Get energy of the state
286  double energy = state.isArtificial() ? 0 : state.getEnergy(cache);
287 
288  // In case of artificial states, symmetries won't work
289  auto sym_reflection_local = sym_reflection;
290  auto sym_rotation_local = sym_rotation;
291  if (state.isArtificial()) {
292  if (sym_reflection_local != NA ||
293  sym_rotation_local.count(static_cast<float>(ARB)) == 0) {
294  std::cerr
295  << "WARNING: Only permutation symmetry can be applied to artificial states."
296  << std::endl;
297  }
298  sym_reflection_local = NA;
299  sym_rotation_local = std::set<float>({static_cast<float>(ARB)});
300  }
301 
302  // Consider rotation symmetry
303  if (sym_rotation_local.count(static_cast<float>(ARB)) == 0 &&
304  sym_rotation_local.count(state.getM()) == 0) {
305  continue;
306  }
307 
308  // Check whether reflection symmetry can be realized with the states available
309  if (sym_reflection_local != NA && state.getM() != 0) {
310  auto state_reflected = state.getReflected();
311  if (this->states_to_add.find(state_reflected) == this->states_to_add.end()) {
312  throw std::runtime_error("The state " + state_reflected.str() +
313  " required by symmetries cannot be found.");
314  }
315  }
316 
317  // Add symmetrized basis vectors
318  this->addSymmetrizedBasisvectors(state, idx, energy, basisvectors_triplets,
319  hamiltonian_triplets, sym_reflection_local);
320  }
321 
322  /// Build data /////////////////////////////////////////////////////
323 
324  this->basisvectors.resize(this->states.size(), idx);
325  this->basisvectors.setFromTriplets(basisvectors_triplets.begin(), basisvectors_triplets.end());
326  basisvectors_triplets.clear();
327 
328  this->hamiltonian.resize(idx, idx);
329  this->hamiltonian.setFromTriplets(hamiltonian_triplets.begin(), hamiltonian_triplets.end());
330  hamiltonian_triplets.clear();
331 }

337 template <typename Scalar>
338 void SystemOne<Scalar>::initializeInteraction() {
339  ////////////////////////////////////////////////////////////////////
340  /// Prepare the calculation of the interaction /////////////////////
341  ////////////////////////////////////////////////////////////////////
342 
343  auto &cache = *this->m_cache;
344 
345  // Check if something to do
346  double tolerance = 1e-24;
347 
348  std::vector<int> erange, brange;
349  std::vector<std::array<int, 2>> drange;
350  std::vector<int> orange;
351  for (const auto &entry : efield_spherical) {
352  if (entry.first < 0) {
353  continue;
354  }
355  if (std::abs(entry.second) > tolerance &&
356  interaction_efield.find(-entry.first) == interaction_efield.end()) {
357  erange.push_back(entry.first);
358  }
359  }
360  for (const auto &entry : bfield_spherical) {
361  if (entry.first < 0) {
362  continue;
363  }
364  if (std::abs(entry.second) > tolerance &&
365  interaction_bfield.find(-entry.first) == interaction_bfield.end()) {
366  brange.push_back(entry.first);
367  }
368  }
369  for (const auto &entry : diamagnetism_terms) {
370  if (entry.first[1] < 0) {
371  continue;
372  }
373  if (diamagnetism && std::abs(entry.second) > tolerance &&
374  interaction_diamagnetism.find(entry.first) == interaction_diamagnetism.end()) {
375  drange.push_back(entry.first);
376  }
377  }
378 
379  if (charge != 0) {
380  for (unsigned int order = 1; order <= ordermax; ++order) {
381  if (interaction_multipole.find(order) == interaction_multipole.end()) {
382  orange.push_back(order);
383  }
384  }
385  }
386  // Return if there is nothing to do
387  if (erange.empty() && brange.empty() && drange.empty() && orange.empty()) {
388  return;
389  }
390 
391  // Precalculate matrix elements
392  auto states_converted = this->getStates();
393  for (const auto &i : erange) {
394  cache.precalculateElectricMomentum(states_converted, i);
395  if (i != 0) {
396  cache.precalculateElectricMomentum(states_converted, -i);
397  }
398  }
399  for (const auto &i : brange) {
400  cache.precalculateMagneticMomentum(states_converted, i);
401  if (i != 0) {
402  cache.precalculateMagneticMomentum(states_converted, -i);
403  }
404  }
405  for (const auto &i : drange) {
406  cache.precalculateDiamagnetism(states_converted, i[0], i[1]);
407  if (i[1] != 0) {
408  cache.precalculateDiamagnetism(states_converted, i[0], -i[1]);
409  }
410  }
411  if (charge != 0) {
412  for (unsigned int order = 1; order <= ordermax; ++order) {
413  cache.precalculateMultipole(states_converted, order);
414  }
415  }
416 
417  ////////////////////////////////////////////////////////////////////
418  /// Calculate the interaction in the canonical basis ///////////////
419  ////////////////////////////////////////////////////////////////////
420 
421  std::unordered_map<int, std::vector<Eigen::Triplet<Scalar>>>
422  interaction_efield_triplets; // TODO reserve
423  std::unordered_map<int, std::vector<Eigen::Triplet<Scalar>>>
424  interaction_bfield_triplets; // TODO reserve
425  std::unordered_map<std::array<int, 2>, std::vector<Eigen::Triplet<Scalar>>,
426  utils::hash<std::array<int, 2>>>
427  interaction_diamagnetism_triplets; // TODO reserve
428  std::unordered_map<int, std::vector<Eigen::Triplet<Scalar>>>
429  interaction_multipole_triplets; // TODO reserve
430  // Loop over column entries
431  for (const auto &c : this->states) { // TODO parallelization
432  if (c.state.isArtificial()) {
433  continue;
434  }
435 
436  // Loop over row entries
437  for (const auto &r : this->states) {
438  if (r.state.isArtificial()) {
439  continue;
440  }
441 
442  // E-field interaction
443  for (const auto &i : erange) {
444  if (i == 0 && r.idx < c.idx) {
445  continue;
446  }
447 
448  if (selectionRulesMultipoleNew(r.state, c.state, 1, i)) {
449  Scalar value = cache.getElectricDipole(r.state, c.state);
450  this->addTriplet(interaction_efield_triplets[i], r.idx, c.idx, value);
451  break; // because for the other operators, the selection rule for the magnetic
452  // quantum numbers will not be fulfilled
453  }
454  }
455 
456  // B-field interaction
457  for (const auto &i : brange) {
458  if (i == 0 && r.idx < c.idx) {
459  continue;
460  }
461 
462  if (selectionRulesMomentumNew(r.state, c.state, i)) {
463  Scalar value = cache.getMagneticDipole(r.state, c.state);
464  this->addTriplet(interaction_bfield_triplets[i], r.idx, c.idx, value);
465  break; // because for the other operators, the selection rule for the magnetic
466  // quantum numbers will not be fulfilled
467  }
468  }
469 
470  // Diamagnetic interaction
471  for (const auto &i : drange) {
472  if (i[1] == 0 && r.idx < c.idx) {
473  continue;
474  }
475 
476  if (selectionRulesMultipoleNew(r.state, c.state, i[0], i[1])) {
477  Scalar value = 1. / (8 * electron_rest_mass) *
478  cache.getDiamagnetism(r.state, c.state, i[0]);
479  this->addTriplet(interaction_diamagnetism_triplets[i], r.idx, c.idx, value);
480  }
481  }
482 
483  // Multipole interaction with an ion
484  if (charge != 0) {
485  int q = r.state.getM() - c.state.getM();
486  if (q == 0) { // total momentum consreved
487  for (const auto &order : orange) {
488  if (selectionRulesMultipoleNew(r.state, c.state, order)) {
489  double val = -coulombs_constant * elementary_charge *
490  cache.getElectricMultipole(r.state, c.state, order);
491  this->addTriplet(interaction_multipole_triplets[order], r.idx, c.idx,
492  val);
493  }
494  }
495  }
496  }
497  }
498  }
499  ////////////////////////////////////////////////////////////////////
500  /// Build and transform the interaction to the used basis //////////
501  ////////////////////////////////////////////////////////////////////
502 
503  for (const auto &i : erange) {
504  interaction_efield[i].resize(this->states.size(), this->states.size());
505  interaction_efield[i].setFromTriplets(interaction_efield_triplets[i].begin(),
506  interaction_efield_triplets[i].end());
507  interaction_efield_triplets[i].clear();
508 
509  if (i == 0) {
510  interaction_efield[i] = this->basisvectors.adjoint() *
511  interaction_efield[i].template selfadjointView<Eigen::Lower>() * this->basisvectors;
512  } else {
513  interaction_efield[i] =
514  this->basisvectors.adjoint() * interaction_efield[i] * this->basisvectors;
515  interaction_efield[-i] = std::pow(-1, i) * interaction_efield[i].adjoint();
516  }
517  }
518 
519  for (const auto &i : brange) {
520  interaction_bfield[i].resize(this->states.size(), this->states.size());
521  interaction_bfield[i].setFromTriplets(interaction_bfield_triplets[i].begin(),
522  interaction_bfield_triplets[i].end());
523  interaction_bfield_triplets[i].clear();
524 
525  if (i == 0) {
526  interaction_bfield[i] = this->basisvectors.adjoint() *
527  interaction_bfield[i].template selfadjointView<Eigen::Lower>() * this->basisvectors;
528  } else {
529  interaction_bfield[i] =
530  this->basisvectors.adjoint() * interaction_bfield[i] * this->basisvectors;
531  interaction_bfield[-i] = std::pow(-1, i) * interaction_bfield[i].adjoint();
532  }
533  }
534 
535  for (const auto &i : drange) {
536  interaction_diamagnetism[i].resize(this->states.size(), this->states.size());
537  interaction_diamagnetism[i].setFromTriplets(interaction_diamagnetism_triplets[i].begin(),
538  interaction_diamagnetism_triplets[i].end());
539  interaction_diamagnetism_triplets[i].clear();
540 
541  if (i[1] == 0) {
542  interaction_diamagnetism[i] = this->basisvectors.adjoint() *
543  interaction_diamagnetism[i].template selfadjointView<Eigen::Lower>() *
544  this->basisvectors;
545  } else {
546  interaction_diamagnetism[i] =
547  this->basisvectors.adjoint() * interaction_diamagnetism[i] * this->basisvectors;
548  interaction_diamagnetism[{{i[0], -i[1]}}] =
549  std::pow(-1, i[1]) * interaction_diamagnetism[i].adjoint();
550  }
551  }
552  if (charge != 0) {
553  for (const auto &i : orange) {
554  interaction_multipole[i].resize(this->states.size(), this->states.size());
555  interaction_multipole[i].setFromTriplets(interaction_multipole_triplets[i].begin(),
556  interaction_multipole_triplets[i].end());
557  interaction_multipole_triplets[i].clear();
558  if (i == 0) {
559  interaction_multipole[i] = this->basisvectors.adjoint() *
560  interaction_multipole[i].template selfadjointView<Eigen::Lower>() *
561  this->basisvectors;
562  } else {
563  interaction_multipole[i] =
564  this->basisvectors.adjoint() * interaction_multipole[i] * this->basisvectors;
565  interaction_multipole[-i] = std::pow(-1, i) * interaction_multipole[i].adjoint();
566  }
567  }
568  }
569 }

575 template <typename Scalar>
576 void SystemOne<Scalar>::addInteraction() {
577  // Build the total Hamiltonian
578  double tolerance = 1e-24;
579 
580  if (std::abs(efield_spherical[+0]) > tolerance) {
581  this->hamiltonian -= interaction_efield[+0] * efield_spherical[+0];
582  }
583  if (std::abs(efield_spherical[-1]) > tolerance) {
584  this->hamiltonian += interaction_efield[+1] * efield_spherical[-1];
585  }
586  if (std::abs(efield_spherical[+1]) > tolerance) {
587  this->hamiltonian += interaction_efield[-1] * efield_spherical[+1];
588  }
589  if (std::abs(bfield_spherical[+0]) > tolerance) {
590  this->hamiltonian -= interaction_bfield[+0] * bfield_spherical[+0];
591  }
592  if (std::abs(bfield_spherical[-1]) > tolerance) {
593  this->hamiltonian += interaction_bfield[+1] * bfield_spherical[-1];
594  }
595  if (std::abs(bfield_spherical[+1]) > tolerance) {
596  this->hamiltonian += interaction_bfield[-1] * bfield_spherical[+1];
597  }
598 
599  if (diamagnetism && std::abs(diamagnetism_terms[{{0, +0}}]) > tolerance) {
600  this->hamiltonian += interaction_diamagnetism[{{0, +0}}] * diamagnetism_terms[{{0, +0}}];
601  }
602  if (diamagnetism && std::abs(diamagnetism_terms[{{2, +0}}]) > tolerance) {
603  this->hamiltonian -= interaction_diamagnetism[{{2, +0}}] * diamagnetism_terms[{{2, +0}}];
604  }
605  if (diamagnetism && std::abs(diamagnetism_terms[{{2, +1}}]) > tolerance) {
606  this->hamiltonian +=
607  interaction_diamagnetism[{{2, +1}}] * diamagnetism_terms[{{2, +1}}] * std::sqrt(3);
608  }
609  if (diamagnetism && std::abs(diamagnetism_terms[{{2, -1}}]) > tolerance) {
610  this->hamiltonian +=
611  interaction_diamagnetism[{{2, -1}}] * diamagnetism_terms[{{2, -1}}] * std::sqrt(3);
612  }
613  if (diamagnetism && std::abs(diamagnetism_terms[{{2, +2}}]) > tolerance) {
614  this->hamiltonian -=
615  interaction_diamagnetism[{{2, +2}}] * diamagnetism_terms[{{2, +2}}] * std::sqrt(1.5);
616  }
617  if (diamagnetism && std::abs(diamagnetism_terms[{{2, -2}}]) > tolerance) {
618  this->hamiltonian -=
619  interaction_diamagnetism[{{2, -2}}] * diamagnetism_terms[{{2, -2}}] * std::sqrt(1.5);
620  }
621  if (charge != 0 && distance != std::numeric_limits<double>::max()) {
622  for (unsigned int order = 1; order <= ordermax; ++order) {
623  double powerlaw = 1. / std::pow(distance, order + 1);
624  this->hamiltonian += interaction_multipole[order] * charge * powerlaw;
625  }
626  }
627 }

718 template <typename Scalar>
719 void SystemOne<Scalar>::incorporate(SystemBase<Scalar, StateOne> &system) {
720  // Combine parameters
721  if (species != dynamic_cast<SystemOne<Scalar> &>(system).species) {
722  throw std::runtime_error(
723  "The value of the variable 'element' must be the same for both systems.");
724  }
725  if (efield != dynamic_cast<SystemOne<Scalar> &>(system).efield) {
726  throw std::runtime_error("The value of the variable 'distance' must be the same for both "
727  "systems."); // implies that efield_spherical is the same, too
728  }
729  if (bfield != dynamic_cast<SystemOne<Scalar> &>(system).bfield) {
730  throw std::runtime_error("The value of the variable 'angle' must be the same for both "
731  "systems."); // implies that
732  // bfield_spherical
733  // is the same,
734  // too
735  }
736  if (diamagnetism != dynamic_cast<SystemOne<Scalar> &>(system).diamagnetism) {
737  throw std::runtime_error(
738  "The value of the variable 'ordermax' must be the same for both systems.");
739  }
740 
741  // Combine symmetries
742  unsigned int num_different_symmetries = 0;
743  if (sym_reflection != dynamic_cast<SystemOne<Scalar> &>(system).sym_reflection) {
744  sym_reflection = NA;
745  ++num_different_symmetries;
746  }
747  if (!(sym_rotation.size() == dynamic_cast<SystemOne<Scalar> &>(system).sym_rotation.size() &&
748  std::equal(sym_rotation.begin(), sym_rotation.end(),
749  dynamic_cast<SystemOne<Scalar> &>(system).sym_rotation.begin()))) {
750  if (sym_rotation.count(static_cast<float>(ARB)) != 0 ||
751  dynamic_cast<SystemOne<Scalar> &>(system).sym_rotation.count(static_cast<float>(ARB)) !=
752  0) {
753  sym_rotation = {static_cast<float>(ARB)};
754  } else {
755  sym_rotation.insert(dynamic_cast<SystemOne<Scalar> &>(system).sym_rotation.begin(),
756  dynamic_cast<SystemOne<Scalar> &>(system).sym_rotation.end());
757  }
758  ++num_different_symmetries;
759  }
760  if (num_different_symmetries > 1) {
761  std::cerr << "Warning: The systems differ in more than one symmetry. For the combined "
762  "system, the notion of symmetries might be meaningless."
763  << std::endl;
764  }
765 
766  // Clear cached interaction
767  this->deleteInteraction();
768 }

774 template <typename Scalar>
775 void SystemOne<Scalar>::addSymmetrizedBasisvectors(
776  const StateOne &state, size_t &idx, const double &energy,
777  std::vector<Eigen::Triplet<Scalar>> &basisvectors_triplets,
778  std::vector<Eigen::Triplet<Scalar>> &hamiltonian_triplets, parity_t &sym_reflection_local) {
779  // In case of reflection symmetry, skip half of the basis vectors
780  if (sym_reflection_local != NA && state.getM() != 0) {
781  if (state.getM() < 0) {
782  return;
783  }
784  }
785 
786  // Store the energy of the unperturbed one atom state
787  hamiltonian_triplets.emplace_back(idx, idx, energy);
788 
789  // Adapt the normalization if required by symmetries
790  Scalar value = 1;
791  if (sym_reflection_local != NA && state.getM() != 0) {
792  value /= std::sqrt(2);
793  }
794 
795  // Add an entry to the current basis vector
796  this->addBasisvectors(state, idx, value, basisvectors_triplets);
797 
798  // Add further entries to the current basis vector if required by symmetries
799  if (sym_reflection_local != NA && state.getM() != 0) {
800  value *= std::pow(-1, state.getL() + state.getM() - state.getJ()) *
801  utils::imaginary_unit<Scalar>();
802  value *= (sym_reflection_local == EVEN) ? 1 : -1;
803  // S_y is invariant under reflection through xz-plane
804  // TODO is the s quantum number of importance here?
805  this->addBasisvectors(state.getReflected(), idx, value, basisvectors_triplets);
806  }
807 
808  ++idx;
809 }
