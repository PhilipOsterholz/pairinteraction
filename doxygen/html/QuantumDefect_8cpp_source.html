86 void QuantumDefect::setup(sqlite3 *db, std::string const &db_name) {
87  static Cache<Key, Element, Hash> cache;
88  static std::string used_db_name;
89 
90  if (used_db_name != db_name) {
91  used_db_name = db_name;
92  cache.clear(); // Clear cache
93  }
94 
95  Key const key{species, n, l, j};
96  if (auto oe = cache.restore(key)) {
97  e = oe.value(); // Restore cache
98  return; // Return early
99  }
100 
101  std::stringstream ss;
102  sqlite::statement stmt(db);
103  int pot_max_l, ryd_max_l;
104  int pot_l, ryd_l;
105  double ryd_max_j;
106  double ryd_j;
107 
108  // Determine maximal L for model potentials
109  stmt.set("select MAX(L) from model_potential where (element = ?);");
110  stmt.prepare();
111  stmt.bind(1, species);
112  if (stmt.step()) {
113  pot_max_l = stmt.get<int>(0);
114  } else {
115  throw no_potential(*this);
116  }
117 
118  // The l to be used is the minimum of the two below
119  pot_l = std::min(l, pot_max_l);
120 
121  // Determine maximal L for Rydberg-Ritz coefficients
122  stmt.reset();
123  stmt.set("select MAX(L) from rydberg_ritz where (element = ?);");
124  stmt.prepare();
125  stmt.bind(1, species);
126  if (stmt.step()) {
127  ryd_max_l = stmt.get<int>(0);
128  } else {
129  throw no_defect(*this);
130  }
131 
132  // The l to be used is the minimum of the two below
133  ryd_l = std::min(l, ryd_max_l);
134 
135  // Determine maximal J for Rydberg-Ritz coefficients
136  stmt.reset();
137  stmt.set("select MAX(J) from rydberg_ritz where ((element = ?1) and (L = "
138  "?2));");
139  stmt.prepare();
140  stmt.bind(1, species);
141  stmt.bind(2, ryd_l);
142  if (stmt.step()) {
143  ryd_max_j = stmt.get<double>(0);
144  } else {
145  throw no_defect(*this);
146  }
147 
148  // The j to be used is the minimum of the two below
149  ryd_j = std::min(j, ryd_max_j);
150 
151  // Load model potentials from database
152  stmt.reset();
153  stmt.set("select ac,Z,a1,a2,a3,a4,rc from model_potential where ((element "
154  "= ?1) and (L = ?2));");
155  stmt.prepare();
156  stmt.bind(1, species);
157  stmt.bind(2, pot_l);
158  if (stmt.step()) {
159  e.ac = stmt.get<double>(0);
160  e.Z = stmt.get<int>(1);
161  e.a1 = stmt.get<double>(2);
162  e.a2 = stmt.get<double>(3);
163  e.a3 = stmt.get<double>(4);
164  e.a4 = stmt.get<double>(5);
165  e.rc = stmt.get<double>(6);
166  } else {
167  throw no_potential(*this);
168  }
169 
170  // Load Rydberg-Ritz coefficients from database
171  stmt.reset();
172  stmt.set("select d0,d2,d4,d6,d8,Ry from rydberg_ritz where ((element = ?1) "
173  "and (L = ?2) and (J = ?3));");
174  stmt.prepare();
175  stmt.bind(1, species);
176  stmt.bind(2, ryd_l);
177  stmt.bind(3, ryd_j);
178  e.nstar = n;
179  double Ry_inf = 109737.31568525;
180  double Ry;
181  if (stmt.step()) {
182  double d0 = stmt.get<double>(0);
183  double d2 = stmt.get<double>(1);
184  double d4 = stmt.get<double>(2);
185  double d6 = stmt.get<double>(3);
186  double d8 = stmt.get<double>(4);
187  Ry = stmt.get<double>(5);
188  e.nstar -= d0 + d2 / std::pow(n - d0, 2) + d4 / std::pow(n - d0, 4) +
189  d6 / std::pow(n - d0, 6) + d8 / std::pow(n - d0, 8);
190  } else {
191  throw no_defect(*this);
192  }
193 
194  e.energy = -.5 * (Ry / Ry_inf) / (e.nstar * e.nstar) * au2GHz;
195 
196  cache.save(key, e);
197 }
