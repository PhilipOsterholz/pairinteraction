124  template <class T>
125  void addRotated(const StateTwo &state, const size_t &idx,
126  std::vector<Eigen::Triplet<T>> &triplets, WignerD &wigner, const double &alpha,
127  const double &beta, const double &gamma) {
128  // Check whether the angles are compatible to the used data type
129  double tolerance = 1e-16;
130  if (std::is_same<T, double>::value &&
131  std::abs(std::remainder(alpha, 2 * M_PI)) > tolerance) {
132  throw std::runtime_error(
133  "If the Euler angle alpha is not a multiple of 2 pi, the Wigner D matrix element "
134  "is complex and cannot be converted to double.");
135  }
136  if (std::is_same<T, double>::value &&
137  std::abs(std::remainder(gamma, 2 * M_PI)) > tolerance) {
138  throw std::runtime_error(
139  "If the Euler angle gamma is not a multiple of 2 pi, the Wigner D matrix element "
140  "is complex and cannot be converted to double.");
141  }
142 
143  // Add rotated triplet entries
144  std::vector<T> val2_vector;
145  val2_vector.reserve(2 * state.getSecondState().getJ() + 1);
146 
147  for (float m2 = -state.getSecondState().getJ(); m2 <= state.getSecondState().getJ(); ++m2) {
148  val2_vector.push_back(
149  utils::convert<T>(wigner(state.getSecondState().getJ(), m2,
150  state.getSecondState().getM(), -gamma, -beta, -alpha)));
151  }
152 
153  for (float m1 = -state.getFirstState().getJ(); m1 <= state.getFirstState().getJ(); ++m1) {
154  auto val1 =
155  utils::convert<T>(wigner(state.getFirstState().getJ(), m1,
156  state.getFirstState().getM(), -gamma, -beta, -alpha));
157 
158  for (float m2 = -state.getSecondState().getJ(); m2 <= state.getSecondState().getJ();
159  ++m2) {
160  StateTwo newstate(state.getSpecies(), state.getN(), state.getL(), state.getJ(),
161  {{m1, m2}});
162  auto state_iter = this->states.template get<1>().find(newstate);
163 
164  if (state_iter != this->states.template get<1>().end()) {
165  auto val = val1 * val2_vector[m2 + state.getSecondState().getJ()];
166  triplets.push_back(Eigen::Triplet<T>(state_iter->idx, idx, val));
167  } else {
168  std::cerr << "Warning: Incomplete rotation because the basis is lacking some "
169  "Zeeman levels."
170  << std::endl;
171  }
172  }
173  }
174  }
