119  template <class T>
120  void addRotated(const StateOne &state, const size_t &idx,
121  std::vector<Eigen::Triplet<T>> &triplets, WignerD &wigner, const double &alpha,
122  const double &beta, const double &gamma) {
123  // Check whether the angles are compatible to the used data type
124  double tolerance = 1e-16;
125  if (std::is_same<T, double>::value &&
126  std::abs(std::remainder(alpha, 2 * M_PI)) > tolerance) {
127  throw std::runtime_error(
128  "If the Euler angle alpha is not a multiple of 2 pi, the Wigner D matrix element "
129  "is complex and cannot be converted to double.");
130  }
131  if (std::is_same<T, double>::value &&
132  std::abs(std::remainder(gamma, 2 * M_PI)) > tolerance) {
133  throw std::runtime_error(
134  "If the Euler angle gamma is not a multiple of 2 pi, the Wigner D matrix element "
135  "is complex and cannot be converted to double.");
136  }
137 
138  // Add rotated triplet entries
139  for (float m = -state.getJ(); m <= state.getJ(); ++m) {
140  StateOne newstate(state.getSpecies(), state.getN(), state.getL(), state.getJ(), m);
141  auto state_iter = this->states.template get<1>().find(newstate);
142 
143  if (state_iter != this->states.template get<1>().end()) {
144  // We calculate the matrix element <m|d_mM|state.getM()>|m>. Note that we must
145  // invert the angles since they are given for a passive rotation, i.e. rotation of
146  // the coordinate system, and the WignerD matrix is for an active rotation of spins.
147  auto val =
148  utils::convert<T>(wigner(state.getJ(), m, state.getM(), -gamma, -beta, -alpha));
149  triplets.push_back(Eigen::Triplet<T>(state_iter->idx, idx, val));
150  } else {
151  std::cerr << "Warning: Incomplete rotation because the basis is lacking some "
152  "Zeeman levels."
153  << std::endl;
154  }
155  }
156  }
