230 Eigen::VectorX<double>
231 SystemBase<Scalar, State>::getOverlap(const std::vector<State> &generalizedstates, double alpha,
232  double beta, double gamma) {
233  // Build basis
234  this->buildBasis();
235 
236  // Determine indices of the specified states
237  std::vector<size_t> states_indices;
238  states_indices.reserve(generalizedstates.size());
239 
240  for (const auto &searched_state : generalizedstates) {
241  if (utils::is_true(searched_state.isGeneralized())) {
242  for (const auto &state : states) {
243  if (state.state ^ searched_state) { // Check whether state.state is contained in
244  // searched_state
245  states_indices.push_back(state.idx);
246  }
247  }
248  } else {
249  auto state_iter = states.template get<1>().find(searched_state);
250  if (state_iter != states.template get<1>().end()) {
251  states_indices.push_back(state_iter->idx);
252  }
253  }
254  }
255 
256  // Get the overlap
257  return this->getOverlap(states_indices, alpha, beta, gamma);
258 }

260 template <class Scalar, class State>
261 Eigen::VectorX<double>
262 SystemBase<Scalar, State>::getOverlap(const std::vector<size_t> &states_indices, double alpha,
263  double beta, double gamma) {
264  // Build basis
265  this->buildBasis();
266 
267  // Build state vectors
268  Eigen::SparseMatrix<Scalar> overlap_states;
269  if (alpha == 0 && beta == 0 && gamma == 0) {
270  std::vector<Eigen::Triplet<Scalar>> overlap_states_triplets;
271  overlap_states_triplets.reserve(states_indices.size());
272 
273  size_t current = 0;
274  for (auto const &idx : states_indices) {
275  overlap_states_triplets.emplace_back(idx, current++, 1);
276  }
277 
278  overlap_states.resize(states.size(), states_indices.size());
279  overlap_states.setFromTriplets(overlap_states_triplets.begin(),
280  overlap_states_triplets.end());
281  } else {
282  overlap_states = this->rotateStates(states_indices, alpha, beta, gamma);
283  }
284 
285  // Calculate overlap
286  Eigen::SparseMatrix<Scalar> product = basisvectors.adjoint() * overlap_states;
287  Eigen::VectorX<double> overlap = Eigen::VectorX<double>::Zero(product.rows());
288  for (int k = 0; k < product.outerSize(); ++k) {
289  for (typename Eigen::SparseMatrix<Scalar>::InnerIterator triple(product, k); triple;
290  ++triple) {
291  overlap[triple.row()] += std::pow(std::abs(triple.value()), 2);
292  }
293  }
294 
295  return overlap;
296 }

390 template <class Scalar, class State>
391 std::array<std::vector<size_t>, 2>
392 SystemBase<Scalar, State>::getConnections(SystemBase<Scalar, State> &system_to, double threshold) {
393  // Build basis
394  this->buildBasis();
395  system_to.buildBasis();
396 
397  double threshold_sqrt = std::sqrt(threshold);
398 
399  // Calculate transformator between the set of states
400  std::vector<Eigen::Triplet<Scalar>> triplets_transformator;
401  triplets_transformator.reserve(std::min(states.size(), system_to.states.size()));
402 
403  for (const auto &s : system_to.states) {
404  auto state_iter = states.template get<1>().find(s.state);
405  if (state_iter != states.template get<1>().end()) {
406  size_t idx_from = state_iter->idx;
407  triplets_transformator.emplace_back(idx_from, s.idx, 1);
408  }
409  }
410 
411  Eigen::SparseMatrix<Scalar> transformator(states.size(), system_to.states.size());
412  transformator.setFromTriplets(triplets_transformator.begin(), triplets_transformator.end());
413 
414  // Calculate overlap
415  Eigen::SparseMatrix<double> overlap_sqrt =
416  (basisvectors.adjoint() * transformator * system_to.basisvectors).cwiseAbs();
417 
418  // Determine the indices of the maximal values within the rows
419  std::vector<size_t> rows_with_maxval;
420  rows_with_maxval.reserve(overlap_sqrt.cols());
421 
422  for (int k = 0; k < overlap_sqrt.outerSize(); ++k) {
423  double maxval = -1;
424  size_t row_with_maxval;
425 
426  for (typename Eigen::SparseMatrix<double>::InnerIterator triple(overlap_sqrt, k); triple;
427  ++triple) {
428  if (triple.value() > maxval) {
429  row_with_maxval = triple.row();
430  maxval = triple.value();
431  }
432  }
433 
434  rows_with_maxval.push_back(row_with_maxval);
435  }
436 
437  // Determine the maximal values within the columns and construct connections
438  Eigen::SparseMatrix<double> overlap_sqrt_transposed = overlap_sqrt.transpose();
439  std::array<std::vector<size_t>, 2> connections;
440  connections[0].reserve(std::max(overlap_sqrt.rows(), overlap_sqrt.cols()));
441  connections[1].reserve(std::max(overlap_sqrt.rows(), overlap_sqrt.cols()));
442 
443  for (int k = 0; k < overlap_sqrt_transposed.outerSize(); ++k) { // cols
444  double maxval = threshold_sqrt;
445  size_t idx_from;
446  size_t idx_to;
447 
448  for (typename Eigen::SparseMatrix<double>::InnerIterator triple(overlap_sqrt_transposed, k);
449  triple; ++triple) {
450  if (triple.value() > maxval) {
451  idx_from = triple.col();
452  idx_to = triple.row();
453  maxval = triple.value();
454  }
455  }
456 
457  if (maxval > threshold_sqrt && rows_with_maxval[idx_to] == idx_from) {
458  connections[0].push_back(idx_from);
459  connections[1].push_back(idx_to);
460  }
461  }
462 
463  connections[0].shrink_to_fit();
464  connections[1].shrink_to_fit();
465 
466  return connections;
467 }

527  // Check variables for consistency
528  if (((hamiltonian.size() == 0) != states.empty()) ||
529  ((hamiltonian.size() == 0) != (basisvectors.size() == 0))) {
530  throw std::runtime_error("Inconsistent variables at " + std::string(__FILE__) + ":" +
531  std::to_string(__LINE__) + ".");
532  }
533 
534  // In case of no new basis restrictions and already initialized basis, there is nothing to
535  // do
536  if (!states.empty() && states_to_add.empty() && range_n.empty() && range_l.empty() &&
537  range_j.empty() && range_m.empty() && energy_min == std::numeric_limits<double>::lowest() &&
538  energy_max == std::numeric_limits<double>::max()) { // TODO check for new threshold, too
539  return;
540  }
541 
542  // Check whether the basis does not exist
543  if (states.empty()) {
544 
545  // Initialize the basis
546  this->initializeBasis();
547 
548  // Forget basis restrictions as they were applied now
549  this->forgetRestrictions();
550 
551  } else {
552  this->updateEverything();
553  }
554 
555  // Check whether the basis is empty
556  if (basisvectors.rows() == 0) {
557  throw std::runtime_error("The basis contains no states.");
558  }
559  if (basisvectors.cols() == 0) {
560  throw std::runtime_error("The basis contains no vectors.");

569 template <class Scalar, class State>
570 void SystemBase<Scalar, State>::diagonalize(double energy_lower_bound, double energy_upper_bound,
571  double threshold) {
572 #ifdef WITH_INTEL_MKL
573  this->buildHamiltonian();
574 
575  // Check if already diagonal
576  if (checkIsDiagonal(hamiltonian)) {
577  return;
578  }
579 
580  // Estimate number of found eigenvalues
581  std::vector<double> diagonal_max(hamiltonian.outerSize(), 0);
582  std::vector<double> diagonal_val;
583  std::vector<int> diagonal_idx;
584  diagonal_val.reserve(hamiltonian.outerSize());
585  diagonal_idx.reserve(hamiltonian.outerSize());
586  for (int k = 0; k < hamiltonian.outerSize(); ++k) {
587  double val = 0;
588  for (typename Eigen::SparseMatrix<Scalar>::InnerIterator triple(hamiltonian, k); triple;
589  ++triple) {
590  if (triple.row() == triple.col()) {
591  val = std::real(triple.value());
592  } else if (triple.row() != triple.col()) {
593  diagonal_max[k] = std::max(diagonal_max[k], std::abs(triple.value()));
594  }
595  }
596  diagonal_idx.push_back(k);
597  diagonal_val.push_back(val);
598  }
599 
600  MKL_INT m0 = std::count_if(diagonal_val.begin(), diagonal_val.end(),
601  [&energy_lower_bound, &energy_upper_bound](const double &val) {
602  return (val < energy_upper_bound) && (val > energy_lower_bound);
603  }) +
604  std::count_if(diagonal_idx.begin(), diagonal_idx.end(),
605  [&energy_lower_bound, &energy_upper_bound, &diagonal_val,
606  &diagonal_max](const int &i) {
607  return ((diagonal_val[i] >= energy_upper_bound) ||
608  (diagonal_val[i] <= energy_lower_bound)) &&
609  (diagonal_val[i] < energy_upper_bound + diagonal_max[i]) &&
610  (diagonal_val[i] > energy_lower_bound - diagonal_max[i]);
611  });
612 
613  // Inplace conversion of the Hamiltonian to CSR with one-based indexing
614  hamiltonian = hamiltonian.transpose();
615  hamiltonian.makeCompressed();
616  for (int i = 0; i < hamiltonian.rows() + 1; ++i) {
617  hamiltonian.outerIndexPtr()[i] += 1;
618  }
619  for (int i = 0; i < hamiltonian.nonZeros(); ++i) {
620  hamiltonian.innerIndexPtr()[i] += 1;
621  }
622 
623  // Set default parameters for the diagonalization as described at
624  // https://software.intel.com/en-us/mkl-developer-reference-c-extended-eigensolver-input-parameters.
625 
626  std::vector<MKL_INT> fpm(128);
627  feastinit(&fpm[0]);
628  fpm[0] = 1; // enables terminal output
629  fpm[1] = 6; // number of contour points
630  fpm[26] = 0; // disables matrix checker
631  fpm[3] = 5; // maximum number of refinement loops allowed
632  if (threshold != 0) {
633  // Adapt the error trace stopping criteria (10-fpm[2])
634  fpm[2] = std::min(std::round(-std::log10(threshold)), 12.);
635  }
636 
637  // Do the diagonalization
638  {
639  MKL_INT n = hamiltonian.rows(); // size of the matrix
640  MKL_INT m; // will contain the number of eigenvalues
641  std::vector<Scalar> x(m0 * n); // the first m columns will contain the eigenvectors
642  {
643  std::vector<double> e(m0); // will contain the first m eigenvalues
644  {
645  char uplo = 'F'; // full matrix is stored
646  MKL_INT info; // will contain return codes
647  double epsout; // will contain relative error
648  MKL_INT loop; // will contain number of used refinement
649  std::vector<double> res(m0); // will contain the residual errors
650 
651  feast_csrev(&uplo, &n, hamiltonian.valuePtr(), hamiltonian.outerIndexPtr(),
652  hamiltonian.innerIndexPtr(), &fpm[0], &epsout, &loop,
653  &energy_lower_bound, &energy_upper_bound, &m0, &e[0], &x[0], &m,
654  &res[0], &info);
655  if (info != 0) {
656  throw std::runtime_error("Diagonalization with FEAST failed.");
657  }
658  }
659 
660  // Build the new hamiltonian
661  hamiltonian.resize(m, m);
662  hamiltonian.setZero();
663  hamiltonian.reserve(m);
664  for (int idx = 0; idx < m; ++idx) {
665  hamiltonian.insert(idx, idx) = e[idx];
666  }
667  hamiltonian.makeCompressed();
668  }
669 
670  // Transform the basis vectors
671  Eigen::SparseMatrix<Scalar> evecs =
672  Eigen::Map<Eigen::MatrixX<Scalar>>(&x[0], n, m).sparseView();
673  if (threshold == 0) {
674  basisvectors = basisvectors * evecs;
675  } else {
676  basisvectors = (basisvectors * evecs).pruned(threshold, 1);
677  }
678  }
679 #else // WITH_INTEL_MKL
680  (void)energy_lower_bound;
681  (void)energy_upper_bound;
682  (void)threshold;
683  throw std::runtime_error(
684  "The method does not work because the program was compiled without MKL support.");
685 #endif // WITH_INTEL_MKL
686 }

779  utils::hash_combine(hashvalue, triple.value());

834 template <class Scalar, class State>
835 void SystemBase<Scalar, State>::add(SystemBase<Scalar, State> &system) {
836  // --- Build bases ---
837  this->buildBasis();
838  system.buildBasis();
839 
840  size_t size_this = basisvectors.cols();
841  size_t size_other = system.basisvectors.cols();
842 
843  // --- Combine system specific variables ---
844  this->incorporate(system);
845 
846  // --- Combine universal variables ---
847  if (memory_saving != system.memory_saving) {
848  throw std::runtime_error(
849  "The value of the variable 'memory_saving' must be the same for both systems.");
850  }
851  if (is_interaction_already_contained != system.is_interaction_already_contained) {
852  throw std::runtime_error("The value of the variable 'is_interaction_already_contained' "
853  "must be the same for both systems.");
854  }
855  if (is_new_hamiltonian_required != system.is_new_hamiltonian_required) {
856  throw std::runtime_error(
857  "The value of the variable 'is_new_hamiltonian_required' must be the same "
858  "for both systems.");
859  }
860 
861  // --- Combine states and build transformators ---
862  Eigen::SparseMatrix<Scalar> transformator;
863  {
864  std::vector<Eigen::Triplet<Scalar>> transformator_triplets;
865  transformator_triplets.reserve(system.states.size());
866 
867  for (const auto &entry : system.states) {
868  auto state_iter = states.template get<1>().find(entry.state);
869 
870  size_t newidx = states.size();
871  if (state_iter == states.template get<1>().end()) {
872  states.push_back(enumerated_state<State>(newidx, entry.state));
873  } else {
874  newidx = state_iter->idx;
875  }
876 
877  transformator_triplets.emplace_back(newidx, entry.idx, 1);
878  }
879 
880  transformator.resize(states.size(), system.basisvectors.rows());
881  transformator.setFromTriplets(transformator_triplets.begin(), transformator_triplets.end());
882  }
883 
884  // --- Combine basisvectors and basisvectors_unperturbed_cache ---
885  basisvectors.conservativeResize(states.size(), size_this + size_other);
886  basisvectors.rightCols(size_other) = transformator * system.basisvectors;
887 
888  if ((basisvectors_unperturbed_cache.size() != 0) !=
889  (system.basisvectors_unperturbed_cache.size() != 0)) {
890  throw std::runtime_error("Inconsistent variables at " + std::string(__FILE__) + ":" +
891  std::to_string(__LINE__) + ".");
892  }
893 
894  if (basisvectors_unperturbed_cache.size() != 0) {
895  basisvectors_unperturbed_cache.conservativeResize(
896  states.size(),
897  basisvectors_unperturbed_cache.cols() + system.basisvectors_unperturbed_cache.cols());
898  basisvectors_unperturbed_cache.rightCols(system.basisvectors_unperturbed_cache.cols()) =
899  transformator * system.basisvectors_unperturbed_cache;
900  }
901 
902  // --- Check if basis vectors are orthogonal ---
903  Eigen::SparseMatrix<Scalar> tmp =
904  (basisvectors.leftCols(size_this).adjoint() * basisvectors.rightCols(size_other))
905  .pruned(1e-12, 1);
906  if (tmp.nonZeros() != 0) {
907  throw std::runtime_error(
908  "Two systems cannot be combined if their basis vectors are not orthogonal.");
909  }
910 
911  // --- Combine hamiltonian and hamiltonian_unperturbed_cache ---
912  Eigen::SparseMatrix<Scalar> shifter(hamiltonian.rows() + system.hamiltonian.rows(),
913  system.hamiltonian.rows());
914  {
915  std::vector<Eigen::Triplet<Scalar>> shifter_triplets;
916  shifter_triplets.reserve(system.hamiltonian.rows());
917 
918  for (Eigen::Index idx = 0; idx < system.hamiltonian.rows(); ++idx) {
919  shifter_triplets.emplace_back(hamiltonian.rows() + idx, idx, 1);
920  }
921 
922  shifter.setFromTriplets(shifter_triplets.begin(), shifter_triplets.end());
923  }
924 
925  hamiltonian.conservativeResize(hamiltonian.rows() + system.hamiltonian.rows(),
926  hamiltonian.cols() + system.hamiltonian.cols());
927  hamiltonian.rightCols(system.hamiltonian.cols()) = shifter * system.hamiltonian;
928 
929  if ((hamiltonian_unperturbed_cache.size() != 0) !=
930  (system.hamiltonian_unperturbed_cache.size() != 0)) {
931  throw std::runtime_error("Inconsistent variables at " + std::string(__FILE__) + ":" +
932  std::to_string(__LINE__) + ".");
933  }
934 
935  if (hamiltonian_unperturbed_cache.size() != 0) {
936  hamiltonian_unperturbed_cache.conservativeResize(
937  hamiltonian_unperturbed_cache.rows() + system.hamiltonian_unperturbed_cache.rows(),
938  hamiltonian_unperturbed_cache.cols() + system.hamiltonian_unperturbed_cache.cols());
939  hamiltonian_unperturbed_cache.rightCols(system.hamiltonian_unperturbed_cache.cols()) =
940  shifter * system.hamiltonian_unperturbed_cache;
941  }

974 template <class Scalar, class State>
975 void SystemBase<Scalar, State>::applySchriefferWolffTransformation(
976  SystemBase<Scalar, State> &system0) {
977  this->diagonalize();
978  system0.buildHamiltonian();
979 
980  // Check that system, on which applySchriefferWolffTransformation() is called, is unitary
981  if (!this->checkIsUnitary(basisvectors)) {
982  throw std::runtime_error("The system, on which applySchriefferWolffTransformation() is "
983  "called, is not unitary. Call unitarize() on the system.");
984  }
985 
986  // Check that system0, i.e. the unperturbed system, is unitary
987  if (!this->checkIsUnitary(system0.basisvectors)) {
988  throw std::runtime_error(
989  "The unperturbed system passed to applySchriefferWolffTransformation() is not "
990  "unitary. Call unitarize() on the system.");
991  }
992 
993  // Check that the unperturbed system is diagonal
994  if (!this->checkIsDiagonal(system0.hamiltonian)) {
995  throw std::runtime_error("The unperturbed system passed to "
996  "applySchriefferWolffTransformation() is not diagonal.");
997  }
998 
999  // --- Express the basis vectors of system0 as a linearcombination of all states ---
1000 
1001  // Combine states and build transformators
1002  Eigen::SparseMatrix<Scalar> transformator;
1003  {
1004  std::vector<Eigen::Triplet<Scalar>> transformator_triplets;
1005  transformator_triplets.reserve(system0.states.size());
1006 
1007  for (const auto &entry : system0.states) {
1008  auto state_iter = states.template get<1>().find(entry.state);
1009 
1010  // Check that all the states of system0 occur (since we already checked that the
1011  // number of states is the same, this ensures that all the states are the same)
1012  if (state_iter == states.template get<1>().end()) {
1013  throw std::runtime_error(
1014  "The unperturbed system contains states that are not occuring.");
1015  }
1016  size_t newidx = state_iter->idx;
1017 
1018  transformator_triplets.emplace_back(newidx, entry.idx, 1);
1019  }
1020 
1021  transformator.resize(states.size(), system0.states.size());
1022  transformator.setFromTriplets(transformator_triplets.begin(), transformator_triplets.end());
1023  }
1024 
1025  // Get basisvectors of system0
1026  Eigen::SparseMatrix<Scalar> low_energy_basis0 = transformator * system0.basisvectors;
1027 
1028  // --- Find basisvectors which corresponds to the basis vectors of system0 ---
1029 
1030  // Get overlaps between basis vectors
1031  Eigen::VectorX<double> overlap = (basisvectors.adjoint() * low_energy_basis0).cwiseAbs2() *
1032  Eigen::VectorX<double>::Ones(low_energy_basis0.cols());
1033 
1034  // Get indices of the low_energy_basis0.cols() largest entries and build transformator
1035  {
1036  std::vector<int> indices(basisvectors.cols());
1037  std::iota(indices.begin(), indices.end(), 0);
1038  std::nth_element(indices.begin(), indices.begin() + low_energy_basis0.cols(), indices.end(),
1039  [&overlap](int a, int b) { return overlap[a] > overlap[b]; });
1040 
1041  std::vector<Eigen::Triplet<Scalar>> transformator_triplets;
1042  transformator_triplets.reserve(low_energy_basis0.cols());
1043 
1044  for (int i = 0; i < low_energy_basis0.cols(); ++i) {
1045  transformator_triplets.emplace_back(indices[i], i, 1);
1046  }
1047 
1048  transformator.resize(basisvectors.cols(), low_energy_basis0.cols());
1049  transformator.setFromTriplets(transformator_triplets.begin(), transformator_triplets.end());
1050  }
1051 
1052  // Get corresponding basis vectors
1053  Eigen::SparseMatrix<Scalar> low_energy_basis = basisvectors * transformator;
1054 
1055  // --- Perform the Schrieffer Wolff transformation ---
1056 
1057  // Projector on the selected basis vectors (typically low-energy subspace)
1058  Eigen::SparseMatrix<Scalar> projector0 = low_energy_basis0 * low_energy_basis0.adjoint();
1059  Eigen::SparseMatrix<Scalar> projector = low_energy_basis * low_energy_basis.adjoint();
1060 
1061  // Reflection operator which change signes of the selectes basis vectors but act
1062  // trivially on others
1063  Eigen::MatrixX<Scalar> reflection0 =
1064  Eigen::MatrixX<Scalar>::Identity(states.size(), states.size()) - 2 * projector0;
1065  Eigen::MatrixX<Scalar> reflection =
1066  Eigen::MatrixX<Scalar>::Identity(states.size(), states.size()) - 2 * projector;
1067 
1068  // Direct rotator from low_energy_basis to low_energy_basis0
1069  Eigen::SparseMatrix<Scalar> rotator = (reflection0 * reflection).sqrt().sparseView();
1070 
1071  if (std::isnan(std::abs(rotator.coeffRef(0, 0)))) {
1072  throw std::runtime_error(
1073  "Error in calculating the matrix square root. Try to use the picomplex module or "
1074  "increase accuracy by choosing smaller thresholds.");
1075  }
1076 
1077  // Calculate the effective Hamiltonian
1078  transformator = basisvectors.adjoint() * rotator.adjoint() * projector0 * low_energy_basis0;
1079 
1080  this->applyRightsideTransformator(transformator);
1081 }
