117  void mergeComplex(std::vector<storage_double> &real, std::vector<storage_double> &imag,
118  std::vector<T> &complex) {
119  std::vector<storage_double>::iterator real_it, imag_it;
120  complex.reserve(real.size());
121  for (real_it = real.begin(), imag_it = imag.begin(); real_it != real.end();

123  complex.push_back(T(*real_it, *imag_it));
124  }
